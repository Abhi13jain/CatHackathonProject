# Firebase Configuration Template
# Copy this file to .env and fill in your Firebase project details

# Your Firebase project ID
FIREBASE_PROJECT_ID=your-project-id

# Firebase service account private key (from JSON file)
FIREBASE_PRIVATE_KEY="-----BEGIN PRIVATE KEY-----\nYour private key here\n-----END PRIVATE KEY-----"

# Firebase service account client email
FIREBASE_CLIENT_EMAIL=your-service-account@your-project.iam.gserviceaccount.com

# Firebase service account client ID
FIREBASE_CLIENT_ID=your-client-id

# Firebase auth domain
FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com

# Firebase web API key
FIREBASE_WEB_API_KEY=your-web-api-key

# Firebase storage bucket
FIREBASE_STORAGE_BUCKET=your-project.appspot.com

# Firebase messaging sender ID
FIREBASE_MESSAGING_SENDER_ID=your-sender-id

# Firebase app ID
FIREBASE_APP_ID=your-app-id

# Firebase measurement ID (optional)
FIREBASE_MEASUREMENT_ID=your-measurement-id

# API configuration
API_V1_STR=/api/v1
PROJECT_NAME="Smart Rental Tracker"

# Instructions:
# 1. Go to Firebase Console: https://console.firebase.google.com/
# 2. Create a new project or select existing one
# 3. Go to Project Settings > Service Accounts
# 4. Generate new private key (downloads JSON file)
# 5. Copy values from JSON to this template
# 6. Save as .env file in the backend directory
